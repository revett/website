{{- /* Emits a right-sized WebP from an asset path. Params: src, max (width, default 1400), alt, class, lazy. Falls back to a plain if the asset isn't found. */ -}} {{- $src := .src -}} {{- $max := .max | default 1400 -}} {{- $alt := .alt | default "" -}} {{- $class := .class -}} {{- $lazy := .lazy -}} {{- $r := resources.Get (strings.TrimPrefix "/" $src) -}} {{- if $r -}} {{- $spec := "webp q80" -}} {{- if gt $r.Width $max }}{{ $spec = printf "resize %dx webp q80" $max }}{{ end -}} {{- $r = $r.Process $spec -}} {{- else -}} {{- end -}}
